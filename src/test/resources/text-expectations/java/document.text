package io.vlingo.xoom.examples.document;

import java.lang.String;

public final class FlightInformation {
  public final String id;

  public final String aircraftId;

  public final String number;

  public final String tailNumber;

  public final String equipment;

  public final String status;

  public FlightInformation(final String id, final String aircraftId, final String number, final String tailNumber, final String equipment, final String status) {
    this.id = id;
    this.aircraftId = aircraftId;
    this.number = number;
    this.tailNumber = tailNumber;
    this.equipment = equipment;
    this.status = status;
  }
}
