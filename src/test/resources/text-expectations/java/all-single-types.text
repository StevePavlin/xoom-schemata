package Context.event;

import io.vlingo.xoom.lattice.model.DomainEvent;
import java.lang.String;

public final class SalutationHappened extends DomainEvent {
  public final String eventType;

  public final long occurredOn;

  public final int eventVersion;

  public final boolean booleanAttribute;

  public final byte byteAttribute;

  public final char charAttribute;

  public final double doubleAttribute;

  public final float floatAttribute;

  public final int intAttribute;

  public final long longAttribute;

  public final short shortAttribute;

  public final String stringAttribute;

  public SalutationHappened(final boolean booleanAttribute, final byte byteAttribute,
      final char charAttribute, final double doubleAttribute, final float floatAttribute,
      final int intAttribute, final long longAttribute, final short shortAttribute,
      final String stringAttribute) {
    this.eventType = "SalutationHappened";
    this.occurredOn = System.currentTimeMillis();
    this.eventVersion = io.vlingo.xoom.common.version.SemanticVersion.toValue("0.0.1");
    this.booleanAttribute = booleanAttribute;
    this.byteAttribute = byteAttribute;
    this.charAttribute = charAttribute;
    this.doubleAttribute = doubleAttribute;
    this.floatAttribute = floatAttribute;
    this.intAttribute = intAttribute;
    this.longAttribute = longAttribute;
    this.shortAttribute = shortAttribute;
    this.stringAttribute = stringAttribute;
  }
}
